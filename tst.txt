state = {"items": set(), "suspicion": 0, "time": 140, "visited": set()}
current_scene = "intro"

def reset_state():
    state["items"].clear()
    state["suspicion"] = 0
    state["time"] = 140
    state["visited"].clear()

def apply_effects(scene_key):
    if scene_key in state["visited"]:
        return
    state["visited"].add(scene_key)
    effects = scenes[scene_key].get("effects", {})
    if effects.get("add_item"):
        add_item = effects["add_item"]
        if isinstance(add_item, (list, tuple, set)):
            state["items"].update(add_item)
        else:
            state["items"].add(add_item)
    if effects.get("remove_item"):
        remove_item = effects["remove_item"]
        if isinstance(remove_item, (list, tuple, set)):
            for item in remove_item:
                state["items"].discard(item)
        else:
            state["items"].discard(remove_item)
    state["suspicion"] = max(0, state["suspicion"] + effects.get("suspicion", 0))
    state["time"] = max(0, state["time"] + effects.get("time", 0))

def set_scene(scene_key):
    global current_scene
    if scene_key == "intro":
        reset_state()
    current_scene = scene_key
    apply_effects(scene_key)
    update_game_state()

def can_choose(choice):
    required = choice.get("requires")
    if not required:
        return True
    if isinstance(required, (list, tuple, set)):
        return all(item in state["items"] for item in required)
    return required in state["items"]

def update_game_state():
    global current_scene
    if current_scene not in ("caught", "escape"):
        if state["time"] <= 0 or state["suspicion"] >= 100:
            current_scene = "caught"

def draw_text_block(text, rect, font, color):
    words = text.split(" ")
    lines = []
    line = ""
    for word in words:
        test = f"{line} {word}".strip()
        if font.size(test)[0] > rect.width - 20:
            lines.append(line)
            line = word
        else:
            line = test
    if line:
        lines.append(line)
    y = rect.top
    for line in lines:
        rendered = font.render(line, True, color)
        screen.blit(rendered, (rect.left + 10, y))
        y += rendered.get_height() + 8

def draw_scene():
    screen.fill(BLACK)

    overlay = pygame.Surface((WIDTH - 100, HEIGHT - 170))
    overlay.set_alpha(230)
    overlay.fill(GRAY)
    screen.blit(overlay, (50, 50))

    title = large_font.render("Escape the School", True, WHITE)
    screen.blit(title, (WIDTH // 2 - title.get_width() // 2, 65))

    text_rect = pygame.Rect(70, 130, WIDTH - 140, 200)
    draw_text_block(scenes[current_scene]["text"], text_rect, font, WHITE)

    exit_button = pygame.Rect(WIDTH - 170, 60, 140, 42)
    mx, my = pygame.mouse.get_pos()
    exit_color = BUTTON_HOVER if exit_button.collidepoint(mx, my) else BUTTON_BG
    pygame.draw.rect(screen, exit_color, exit_button, border_radius=12)
    exit_text = font.render("Close Game", True, WHITE)
    screen.blit(exit_text, (exit_button.centerx - exit_text.get_width() // 2,
                            exit_button.centery - exit_text.get_height() // 2))

    buttons = [(exit_button, {"exit": True})]
    for idx, choice in enumerate(scenes[current_scene]["choices"]):
        button_rect = pygame.Rect(70, 380 + idx * 70, WIDTH - 140, 52)
        mx, my = pygame.mouse.get_pos()
        available = can_choose(choice)
        hovered = button_rect.collidepoint(mx, my)
        color = BUTTON_HOVER if hovered and available else BUTTON_BG
        if not available:
            color = BUTTON_DISABLED
        pygame.draw.rect(screen, color, button_rect, border_radius=12)
        text_surf = font.render(choice["text"], True, WHITE)
        screen.blit(text_surf, (button_rect.left + 16, button_rect.centery - text_surf.get_height() // 2))
        if not available and choice.get("lock_text"):
            lock_text = font.render(choice["lock_text"], True, (220, 140, 120))
            screen.blit(lock_text, (button_rect.left + 16, button_rect.bottom + 4))
        buttons.append((button_rect, choice))
    return buttons

set_scene("intro")


running = True
while running:
    clock.tick(60)
    update_game_state()
    buttons = draw_scene()
    pygame.display.flip()
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
        elif event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
            for rect, choice in buttons:
                if rect.collidepoint(event.pos) and can_choose(choice):
                    if choice.get("exit"):
                        running = False
                    else:
                        set_scene(choice["next"])
                    break

pygame.quit()
sys.exit()
